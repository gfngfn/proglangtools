\newcommand\meta[1]{\(\langle\mathit{#1}\rangle\)}
\begin{document}
  \section{How to use}
  \begin{itemize}
    \item \texttt{\string\infer[\meta{label}]\{\meta{hyposethes}\}\{\meta{conclusion}\}}:
      typesets a derivation tree.
      \meta{hypotheses} is a \texttt{\string\andalso}-separated list of derivation trees.
  \end{itemize}
  \section{Example}
  \begin{center}
    \infer[Foo]{%
      \infer[Bar]{%
        A
      \andalso
        \infer{}{%
          B
        }
      }{%
        C
      }
    \andalso
      D
    \andalso
      \infer{%
        E
      \andalso
        \infer[Baz]{
          \infer{%
            \infer[Qux]{}{%
              F
            }
          }{%
            G
          }
        \andalso
          H
        }{%
          I
        }
      }{%
        J
      }
    }{%
      K
    }
  \end{center}
